# Путь обычно:
#   C:\ProgramData\MySQL\MySQL Server 9.7\my.ini
#
# 1. Сделайте копию: my.ini.bak
# 2. УДАЛИТЕ или закомментируйте (#) строки, которые вы добавляли вручную
#    (лишние bind-address, skip-networking, datadir с ошибкой и т.д.)
# 3. Для диплома bind-address НЕ нужен — FastAPI -> localhost
#
# Типичные обязательные строки (пути проверьте у себя в Installer):

[mysqld]
# datadir=C:/ProgramData/MySQL/MySQL Server 9.7/Data
# port=3306
